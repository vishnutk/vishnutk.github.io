ES6 String.repeat with console.time()

Jun 30, 2015

Today, I was trying to implement a simple Staircase problem. With implementation 1 I wrote it using two for loops. Its works and I was able to see staircases on the browser. But as I tried to increment n by multiple of 10000, browser stop to respond. Thereafter I used String.prototype.repeat for the same problem. This native method then shows its features. There was a huge performance gain and was able to achive the same result. Just check below two approaches and the time difference. console.time() helps a lot. 787 ms improvement for n=10000

Implementation 1


                // n = 10000
                // processData: 798.195ms
                console.time('processData');
                function processData( n ) {
                    var i, j, str;

                    for( i = 0; i < n; i++ ) {
                        str = '';
                        for( j = n; j > 0; j-- ) {
                            if(j <= i+1) {
                                str = str + '#';
                            } else {
                                str = str + ' ';
                            }

                        }
                        // Below statement is commented explicitly since n is a large number.
                        // console.log( str );
                    }
                }
                processData(10000);
                console.timeEnd('processData');

Implementation 2


                // n=10000
                // processData: 11.759ms

                // n=100000
                // processData: 97.847ms


                console.time('processData');
                function processData( n ) {
                    var i,
                        str;

                    for( i = 1; i <= n; i++ ) {
                        str = ' '.repeat( n-i ) + '#'.repeat( i );
                        // Below statement is commented explicitly since n is a large number.
                        // console.log( str );
                    }
                }
                processData(10000);
                console.timeEnd('processData');
